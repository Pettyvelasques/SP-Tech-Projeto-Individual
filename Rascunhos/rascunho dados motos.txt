total moto
	select count(fkMoto) as totalMotos from motoNaGaragem;

motos compradas
	select count(fkMoto) as totalCompradas from motoNaGaragem where motoComprada=1;

valor garagem
	select sum(m.preco) as precoTotal from motoNaGaragem as mg join moto as m on fkMoto = idMoto where mg.fkGaragem=1;

valor compradas
	select sum(m.preco) as precoCompradas from motoNaGaragem as mg join moto as m on fkMoto = idMoto where mg.fkGaragem=1 and mg.motoComprada=1;
